-رحلة كازوز-

فــي أواخــر 2020 واحــنــا هــاربــيــن مــن طــوفـان الـحــداثـة,وقــفــنــا أمــام كــهــف مــظــلــم وكــبــيــر,وكــان عــلــى مــدخــلــة كــازوز مــعــلــق,فــأخــذنــاه,وإنــطــلقــنــا لـلـكـهـف.. هـناك وعـلـى الـجـدران,عـيـونـنـا شـافـت تـاريـخ نـحـتـه أجـدادنـا, تاريـخ كان بـيـنـتـسـي داخـل كـهـف مـهـجـور.. ومـن تـلـك الـلـحـظـة,قـررنا إن احـنـا نـحـول الـنـقـوش إلى حـكـاية بـصـريـة,ونرجـع للطوفان,ونعتلية ماسـكـيـن تراثنا الضائع لعـالـم كاد حاضـره يطمس ماضيه..فنخلي تاريخنا اليمني أقرب,وعلى نطاق واسع,وفي متناول الجميع.

الــرســالــه-

نــشــعــل الــكــازوز .. ونــحــفــر فــي خــبــايــا الــزمــان لــنــحــيــي الأســطــورة الــيــمــنــيــة ونــخــلــق مــنـــهــا عــالــمــاً مــن الــرســوم .. فـــنـــجــمــع إنــســان الــمــاضــي مــع إنــســان الــيــوم تــحــت مــظــلــة زمــنــيــة واحــدة..

الــرؤيــة-

أن يــرانــا بــاحــث الــتــاريــخ فــيــقــول لــمــن مــعــه إنــي آنــســت نــاراً لــعــلــي آتــيــكــم مــنــهــا بــقــبــس.
